// ── FLAT ACTIONS ──────────────────────────────────────────────────────────

function flatActions(items){

  const result=[];

  items.forEach(item=>{

    (function walk(actions){

      actions.forEach(a=>{

        result.push({action:a,item});

        if(a.children&&a.children.length) walk(a.children);

      });

    })(item.actions||[]);

  });

  return result;

}

// ── TREE ──────────────────────────────────────────────────────────────────

const expandState={};

function renderTree(item){

  const el=document.getElementById('tree-'+item.id); if(!el) return;

  if(!(item.actions||[]).length){el.innerHTML='<div class="empty">Sin acciones. Agregá la primera.</div>';return;}

  el.innerHTML=renderNodes(item.actions,item);

}

function renderNodes(actions,item){ return actions.map(a=>renderNode(a,item)).join(''); }

function renderNode(a,item){

  const hasChildren=a.children&&a.children.length>0;

  const isOpen=expandState[a.id]!==false;

  const st=dateStatus(a.date);

  const dtClass=st==='overdue'?'overdue':st==='soon'?'soon':'';

  const stateColors={pendiente:'var(--hint)',en_curso:'var(--blue)',completado:'var(--green)',bloqueado:'var(--red)'};

  return `<div class="action-node">

    <div class="action-row" style="border-left:2.5px solid ${stateColors[a.estado]||'var(--hint)'}">

      <button class="action-toggle" onclick="toggleExpand('${a.id}','${item.id}')" ${!hasChildren?'style="visibility:hidden"':''}>${hasChildren&&isOpen?'▾':'▸'}</button>

      <input type="checkbox" class="action-check" ${a.done?'checked':''} onchange="toggleDone('${a.id}','${item.id}')">

      <div class="action-content">

        <div class="action-text ${a.done?'done':''}">${a.text}</div>

        <div class="action-dates">

          ${a.date?`<span class="action-date-tag ${dtClass}">📅 Revisión: ${fmtDate(a.date)}</span>`:`<span class="action-date-tag" style="cursor:pointer" onclick="openEditActionModal('${a.id}','${item.id}')">+ Agregar fecha</span>`}

          ${a.responsable?`<span class="action-date-tag">👤 ${a.responsable}</span>`:''}

        </div>

      </div>

      <div class="action-row-btns">

        <button class="btn ghost sm" onclick="openEditActionModal('${a.id}','${item.id}')">✏️ Editar</button>

        <button class="btn ghost sm" onclick="showAddChildForm('${a.id}','${item.id}')">+ Sub-acción</button>

        <button class="btn ghost sm" style="color:var(--red)" onclick="deleteAction('${a.id}','${item.id}')">🗑 Borrar</button>

      </div>

    </div>

    <div id="addform-${a.id}" style="display:none;padding-left:22px;margin-top:3px"></div>

    ${hasChildren&&isOpen?`<div class="action-indent">${renderNodes(a.children,item)}</div>`:''}

  </div>`;

}


window.toggleExpand=function(aid,itemId){ expandState[aid]=expandState[aid]===false?true:false; rerenderTree(itemId); }

window.toggleDone=function(aid,itemId){

  const item=findItem(itemId); const a=findAction(item.actions,aid);

  if(a){ a.done=!a.done; rerenderTree(itemId); window.saveDB(); }

}

function rerenderTree(itemId){ const item=findItem(itemId); if(item) renderTree(item); }

function findItem(id){ return [...(window.DB.proyectos||[]),...(window.DB.desarrollos||[])].find(x=>x.id===id); }

function findAction(actions,id){

  for(const a of actions){

    if(a.id===id) return a;

    if(a.children){ const f=findAction(a.children,id); if(f) return f; }

  }

  return null;

}

function findParentArr(actions,id){

  for(const a of actions){

    if(a.children){

      if(a.children.find(c=>c.id===id)) return a.children;

      const f=findParentArr(a.children,id); if(f) return f;

    }

  }

  return null;

}


window.addRootAction=function(key,itemId){

  const item=findItem(itemId); if(!item) return;

  const na=makeAction('Nueva acción');

  item.actions.push(na);

  rerenderTree(itemId);

  openEditActionModal(na.id,itemId);

}

    <input class="add-action-date" type="date" id="child-date-${parentId}">

    <button class="btn sm primary" onclick="saveChildAction('${parentId}','${itemId}')">Agregar</button>

    <button class="btn sm" onclick="document.getElementById('addform-${parentId}').style.display='none'">✕</button>

  </div>`;

  document.getElementById('child-txt-'+parentId).focus();

}

window.saveChildAction=function(parentId,itemId){

  const txt=document.getElementById('child-txt-'+parentId)?.value.trim(); if(!txt) return;

  const date=document.getElementById('child-date-'+parentId)?.value||'';

  const item=findItem(itemId); const parent=findAction(item.actions,parentId);

  if(!parent) return;

  if(!parent.children) parent.children=[];

  expandState[parentId]=true;

  parent.children.push(makeAction(txt,date));

  document.getElementById('addform-'+parentId).style.display='none';

  rerenderTree(itemId); window.saveDB();

}

window.deleteAction=function(aid,itemId){

  if(!confirm('¿Eliminar esta acción y todas sus sub-acciones?')) return;

  const item=findItem(itemId);

  const parentArr=findParentArr(item.actions,aid)||item.actions;

  const idx=parentArr.findIndex(a=>a.id===aid);

  if(idx>-1){ parentArr.splice(idx,1); rerenderTree(itemId); window.saveDB(); }

}

// ── VENCIMIENTOS ──────────────────────────────────────────────────────────

function renderVencimientos(){

  if(!window.DB) return;

  const allActions=flatActions([...window.DB.proyectos,...window.DB.desarrollos]).filter(x=>x.action.date&&!x.action.done);

  const filters=['todos','overdue','soon','ok'];

  const labels={todos:'Todos',overdue:'Vencidos',soon:'Esta semana',ok:'Futuros'};

  document.getElementById('flt-venc').innerHTML=filters.map(f=>

    `<button class="flt ${filterVenc===f?'on':''}" onclick="setFilterVenc('${f}')">${labels[f]} (${f==='todos'?allActions.length:allActions.filter(x=>dateStatus(x.action.date)===f).length})</button>`

  ).join('');

  const filtered=filterVenc==='todos'?allActions:allActions.filter(x=>dateStatus(x.action.date)===filterVenc);

  filtered.sort((a,b)=>(a.action.date||'9999')>(b.action.date||'9999')?1:-1);

  const el=document.getElementById('list-venc');

  if(!filtered.length){el.innerHTML='<div class="empty">Sin acciones con fecha en esta vista</div>';return;}

  el.innerHTML=filtered.map(({action:a,item:i})=>{

    const st=dateStatus(a.date);

    return `<div class="overdue-item" style="cursor:pointer" onclick="openDetailFromDash('${i.id}')">

      <span class="action-date-tag ${st}">📅 ${fmtDate(a.date)}</span>

      <div style="flex:1;min-width:0"><div style="font-size:12px;font-weight:500">${a.text}</div>

      <div style="font-size:10px;color:var(--muted)">${i.nombre}${a.responsable?' · '+a.responsable:''}</div></div>

      ${estadoBadge(a.estado)}

    </div>`;

  }).join('');

}

window.setFilterVenc=function(v){ filterVenc=v; renderVencimientos(); }

window.saveNewItem=function(){

  const nombre=document.getElementById('fi-nombre').value.trim(); if(!nombre){alert('Ingresá el nombre');return;}

  const tipo=document.getElementById('fi-tipo').value;

  const estado=document.getElementById('fi-estado').value;

  const pg=parseInt(document.getElementById('fi-pg').value)||0;

  const inst=parseInt(document.getElementById('fi-inst').value)||0;

  const {type,editId}=newItemCtx;

  const arr=type==='proyecto'?window.DB.proyectos:window.DB.desarrollos;

  if(editId){

    const item=arr.find(x=>x.id===editId);

    if(item) Object.assign(item,{nombre,tipo,estado,pg,inst});

    closeModalItem();

    const key=type==='proyecto'?'proy':'dev';

    renderDetail(key,editId);

  } else {

    const nums=arr.map(x=>x.num); const num=nums.length?Math.max(...nums)+1:1;

    arr.push({id:uid(),num,nombre,tipo,estado,pg,inst,actions:[]});

    closeModalItem();

    renderList(type==='proyecto'?'proy':'dev');

  }

  window.saveDB();

}


// ── EXPORT EXCEL ──────────────────────────────────────────────────────────

window.exportExcel=function(){

  if(typeof XLSX==='undefined'){alert('Cargando librería, intentá en segundos...');return;}

  if(!window.DB){alert('Sin datos');return;}

  const wb=XLSX.utils.book_new();

  const elbl={en_curso:'En curso',pendiente:'Pendiente',completado:'Completado',bloqueado:'Bloqueado'};

  function walkRows(actions,item,rows,depth){

    actions.forEach(a=>{

      const prefix='  '.repeat(depth)+(depth>0?'↳ ':'');

      rows.push([depth===0?item.num:'',depth===0?item.nombre:'',depth===0?item.tipo:'',depth===0?(elbl[item.estado]||item.estado):'',depth===0?item.pg+'%':'',depth===0?item.inst+'%':'','Nivel '+(depth+1),prefix+a.text,elbl[a.estado]||a.estado||'',a.date?fmtDate(a.date):'',a.responsable||'',a.done?'Sí':'No']);

      if(a.children&&a.children.length) walkRows(a.children,item,rows,depth+1);

    });

  }

  const hdr=['#','Nombre','Tipo','Estado','% PG','% Inst/Pruebas','Nivel','Acción','Estado acción','Fecha revisión','Responsable','Completada'];

  const rp=[hdr]; window.DB.proyectos.forEach(i=>{ if(!i.actions.length) rp.push([i.num,i.nombre,i.tipo,elbl[i.estado]||i.estado,i.pg+'%',i.inst+'%','','','','','','']); else walkRows(i.actions,i,rp,0); });

  const wp=XLSX.utils.aoa_to_sheet(rp); wp['!cols']=[4,30,16,12,8,8,8,52,12,14,16,10].map(w=>({wch:w})); XLSX.utils.book_append_sheet(wb,wp,'Proyectos');

  const rd=[hdr]; window.DB.desarrollos.forEach(i=>{ if(!i.actions.length) rd.push([i.num,i.nombre,i.tipo,elbl[i.estado]||i.estado,i.pg+'%',i.inst+'%','','','','','','']); else walkRows(i.actions,i,rd,0); });

  const wd=XLSX.utils.aoa_to_sheet(rd); wd['!cols']=[4,30,16,12,8,8,8,52,12,14,16,10].map(w=>({wch:w})); XLSX.utils.book_append_sheet(wb,wd,'Desarrollos');

  const allF=flatActions([...window.DB.proyectos,...window.DB.desarrollos]).filter(x=>x.action.date);

  allF.sort((a,b)=>(a.action.date||'9999')>(b.action.date||'9999')?1:-1);

  const rv=[['Proyecto/Desarrollo','Tipo','Acción','Estado','Fecha revisión','Responsable','Completada','Alerta']];

  allF.forEach(({action:a,item:i})=>{ const st=dateStatus(a.date); rv.push([i.nombre,i.tipo,a.text,elbl[a.estado]||a.estado||'',fmtDate(a.date),a.responsable||'',a.done?'Sí':'No',st==='overdue'?'VENCIDA':st==='soon'?'Esta semana':'OK']); });

  const wv=XLSX.utils.aoa_to_sheet(rv); wv['!cols']=[30,16,52,12,14,16,10,12].map(w=>({wch:w})); XLSX.utils.book_append_sheet(wb,wv,'Vencimientos');

  const allI=[...window.DB.proyectos,...window.DB.desarrollos]; const allA=flatActions(allI);

  const rr=[['RESUMEN – COMECA ARG / PROYECTOS I&D 2026'],['Exportado:',new Date().toLocaleDateString('es-AR')],[],['ÍTEM','VALOR'],['Proyectos activos',window.DB.proyectos.length],['Desarrollos activos',window.DB.desarrollos.length],['Total',allI.length],['En curso',allI.filter(i=>i.estado==='en_curso').length],['Completados',allI.filter(i=>i.estado==='completado').length],['Pendientes/Bloqueados',allI.filter(i=>i.estado==='pendiente'||i.estado==='bloqueado').length],[],['Total acciones',allA.length],['Completadas',allA.filter(x=>x.action.done).length],['Con fecha de revisión',allA.filter(x=>x.action.date).length],['Vencidas',allA.filter(x=>!x.action.done&&dateStatus(x.action.date)==='overdue').length],['Próximas (7 días)',allA.filter(x=>!x.action.done&&dateStatus(x.action.date)==='soon').length]];

  const wr=XLSX.utils.aoa_to_sheet(rr); wr['!cols']=[{wch:32},{wch:16}]; XLSX.utils.book_append_sheet(wb,wr,'Resumen');

  XLSX.writeFile(wb,'ProyectosID_'+new Date().toISOString().slice(0,10)+'.xlsx');

}
